# CreateOAuthAppRequest

Request to create a new OAuth app

## Example Usage

```typescript
import { CreateOAuthAppRequest } from "@pipeshub/sdk/models";

let value: CreateOAuthAppRequest = {
  name: "My Integration App",
  description: "Integrates PipesHub with our internal tools",
  redirectUris: [
    "https://myapp.com/callback",
    "http://localhost:3000/callback",
  ],
  allowedGrantTypes: [
    "authorization_code",
    "refresh_token",
  ],
  allowedScopes: [
    "openid",
    "profile",
    "read:records",
  ],
};
```
